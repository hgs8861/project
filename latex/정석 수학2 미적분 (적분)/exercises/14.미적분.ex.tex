\section*{기본 문제}

\begin{enumerate}
    \item \textbf{Exercise 14-1} \\
    다음 부정적분을 구하여라.
    \begin{enumerate}[label=(\arabic*)]
        \item $\int\dfrac{x}{(2x+1)^{3}}dx$
        \item $\int\dfrac{x}{\sqrt{1-x^{2}}}dx$
        \item $\int x\sin x^{2}dx$
        \item $\int\dfrac{e^{x}-1}{e^{x}+1}dx$
        \item $\int\dfrac{\ln x}{x}dx$
    \end{enumerate}

    \item \textbf{Exercise 14-3} \\
    함수 $f(x), g(x)$ 가 다음 세 조건을 만족시킬 때, $f(x)+g(x)$ 를 구하여라.
    \begin{enumerate}[label=(\text{\roman*})]
        \item $f^{\prime}(x)=2g(x), g^{\prime}(x)=2f(x)$
        \item $f(0)=1, g(0)=e-1$
        \item $f(x)>0, g(x)>0$
    \end{enumerate}

    \item \textbf{Exercise 14-5} \\
    미분가능한 함수 $f(x)$ 가 모든 실수 $x$ 에 대하여 $xf^{\prime}(x)-f(x)=(x^{4}+2x^{2})e^{x-1}$ 을 만족시킨다. $f(1)=3$ 일 때, $f(3)$ 의 값을 구하여라.
\end{enumerate}

\section*{실력 문제}

\begin{enumerate}
    \item \textbf{Exercise 14-6} \\
    다음 부정적분을 구하여라.
    \begin{enumerate}[label=(\arabic*)]
        \item $\int\dfrac{x^{4}}{(x-1)^{3}}dx$
        \item $\int\sqrt{1+\sqrt{x}}dx$
    \end{enumerate}

    \item \textbf{Exercise 14-7} \\
    다음 부정적분을 구하여라.
    \begin{enumerate}[label=(\arabic*)]
        \item $\int\dfrac{1}{\sin x}dx$
        \item $\int\dfrac{\sin x}{1+\sin x}dx$
    \end{enumerate}

    \item \textbf{Exercise 14-11} \\
    부정적분 $\int x\cot^{2}x~dx$ 를 구하여라.

    \item \textbf{Exercise 14-14} \\
    다음 등식이 성립함을 증명하여라. (단, $n$ 은 2 이상의 자연수)
    \begin{enumerate}[label=(\arabic*)]
        \item $\int \tan^{n}x~dx=\dfrac{\tan^{n-1}x}{n-1}-\int \tan^{n-2}x~dx$
        \item $\int \sin^{n}x~dx=-\dfrac{\sin^{n-1}x\cos x}{n}+\dfrac{n-1}{n}\int \sin^{n-2}x~dx$
    \end{enumerate}
\end{enumerate}
