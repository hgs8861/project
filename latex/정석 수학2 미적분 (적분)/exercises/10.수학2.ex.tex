\section*{기본 문제}

\begin{enumerate}
    \item \textbf{Exercise 10-1} \\
    다음 정적분의 값을 구하여라.
    \[ \int_{1}^{2} \left( 3x^2 - \dfrac{4}{x^2} + 1 \right) dx \]

    \item \textbf{Exercise 10-2} \\
    함수 $f(x) = \begin{cases} x^2 - x & (x \ge 1) \\ x - 1 & (x < 1) \end{cases}$ 일 때, $\int_{0}^{2} f(x) dx$의 값을 구하여라.

    \item \textbf{Exercise 10-3} \\
    $\int_{0}^{2} |x^2 - x| dx$의 값을 구하여라.

    \item \textbf{Exercise 10-4} \\
    다항함수 $f(x)$가 임의의 실수 $x$에 대하여 $\int_{1}^{x} f(t) dt = x^3 - 2x^2 + ax$를 만족시킬 때, 상수 $a$의 값과 $f(2)$의 값을 구하여라.

    \item \textbf{Exercise 10-5} \\
    다음 정적분의 값을 구하여라.
    \[ \int_{-2}^{2} (5x^7 - 3x^5 + 4x^3 - x + 3) dx \]
\end{enumerate}

\section*{실력 문제}

\begin{enumerate}
    \item \textbf{Exercise 10-6} \\
    다항함수 $f(x)$가 모든 실수 $x$에 대하여 다음 등식을 만족시킬 때, $f(1)$의 값을 구하여라.
    \[ f(x) = x^2 - 3x + \int_{0}^{2} f(t) dt \]

    \item \textbf{Exercise 10-7} \\
    함수 $f(x) = \int_{0}^{x} (t^2 - 2t - 3) dt$가 $x=a$에서 극댓값, $x=b$에서 극솟값을 가질 때, $b-a$의 값을 구하여라.

    \item \textbf{Exercise 10-8} \\
    임의의 실수 $x$에 대하여 $\int_{1}^{x} (x-t)f(t) dt = x^3 + ax^2 + bx + 1$이 성립할 때, 상수 $a, b$의 값과 $f(x)$를 구하여라.

    \item \textbf{Exercise 10-9} \\
    다음 극한값을 구하여라.
    \[ \lim_{x \to 2} \dfrac{1}{x-2} \int_{2}^{x} (t^3 - 3t^2 + 4) dt \]

    \item \textbf{Exercise 10-10} \\
    연속함수 $f(x)$가 모든 실수 $x$에 대하여 $f(x+2) = f(x)$이고, $\int_{-1}^{1} f(x) dx = 4$를 만족시킬 때, $\int_{-1}^{5} f(x) dx$의 값을 구하여라.
\end{enumerate}
