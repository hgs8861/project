    % [기본 문제 섹션 - 4문항 선정 (전체 8문항의 절반)]
    \exercise{$f(x)=\int_{0}^{x} \frac{x-t}{\cos^2 t} \,dt$ $\left(\text{단, } -\frac{\pi}{2} < x < \frac{\pi}{2}\right)$일 때, $0 \le x \le \frac{\pi}{6}$에서 곡선 $y=f(x)$의 길이를 구하여라.} 
    & 
    \exercise{수직선 위를 움직이는 점 P의 시각 $t$ 에서의 속도 $v(t)$ 는 $v(t) = \cos t + \cos 2t$ 이고, $t=0$ 일 때 점 P는 원점에 있다. 
    \begin{enumerate}[label=(\arabic*), leftmargin=8mm, nosep]
        \item 시각 $t$ 에서의 점 P의 위치를 $t$ 로 나타내어라.
        \item $t=0$ 일 때부터 $t=\pi$ 일 때까지 점 P가 움직인 거리를 구하여라.
    \end{enumerate}} \\
    \hline 
    
    \exercise{다음 주어진 구간에서 곡선의 길이를 구하여라.
    \begin{enumerate}[label=(\arabic*), leftmargin=8mm, nosep]
        \item $y = x\sqrt{x} (0 \le x \le 2)$
        \item $y = \ln(\sec x) (0 \le x \le \dfrac{\pi}{4})$
    \end{enumerate}} 
    & 
    \exercise{좌표평면 위를 움직이는 점 $\text{P}$의 시각 $t$ (단, $t \ge 0$)에서의 위치 $(x, y)$가 $x = 1 + \frac{5}{4}t^2$, $y = 1 + t^{\frac{5}{2}}$일 때, 다음 물음에 답하여라.
    \begin{enumerate}
        \item[(1)] 점 $\text{P}$가 그리는 곡선의 접선의 기울기가 $1$이 되는 시각 $t_0$을 구하여라.
        \item[(2)] (1)의 $t_0$에 대하여 $t=0$일 때부터 $t=t_0$일 때까지 점 $\text{P}$가 움직인 거리를 구하여라.
    \end{enumerate}} \\
    \hline

    % [실력 문제 섹션 - 3문항 선정 (전체 6문항의 절반)]
    \exercise{좌표평면 위의 점 $\text{P}$는 점 $(0, 1)$을 출발하여 매초 $1$의 속력으로 제1사분면에 있는 곡선 $y=\frac{1}{2}(e^x+e^{-x})$ 위를 움직인다. $t$초 후 점 $\text{P}$의 $x$좌표를 $t$로 나타내어라.} 
    & 
    \exercise{좌표평면 위에서 직선 $g$가 원 $x^2+y^2=1$과 점 $\text{A}(1, 0)$에서 접하고 있다. 직선 $g$가 원에 접하면서 미끄러지지 않게 시계 반대 방향으로 $90^\circ$ 회전할 때, 점 $\text{A}$와 일치해 있던 직선 $g$ 위의 점 $\text{P}$가 그리는 곡선의 길이를 구하여라.} \\
    \hline

    \exercise{실수 전체의 집합에서 이계도함수를 가지는 함수 $f(t)$ 에 대하여 좌표평면 위를 움직이는 점 P의 시각 $t$ 에서의 위치 $(x, y)$ 가 $x = 4\cos t, y = f(t)$ 이다. 점 P가 $t=0$ 일 때부터 $t=s(s > 0)$ 일 때까지 움직인 거리는 $\dfrac{9}{2}s - \dfrac{\sin 2s}{4}$ 이고, $t=\pi$ 일 때 점 P의 속도는 $(0, 4)$ 이다. $t=\dfrac{\pi}{4}$ 일 때, 점 P의 가속도의 크기를 구하여라.}
