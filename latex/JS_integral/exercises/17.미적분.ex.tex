        \item $y = \tan x, \quad x = \frac{\pi}{3}, \quad y = 0$
        \item $y = (1 - \cos x)\sin x \quad (0 \le x \le \pi), \quad y = 0$       
        \item $y = e^x - 1, \quad y = 4e^{-x} - 1, \quad y = 0$        
        \item $y = \sqrt{-x+1}, \quad y = 0, \quad y = 2, \quad x = 0$  
        \item $y = \ln(2-x), \quad y = 0, \quad x = 0$      
        \item $y^2 = 2x, \quad x^2 = 2y$
    \end{enumerate}}

    \exercise{미분가능한 함수 $f(x)$가 모든 실수 $x$에 대하여 $f(x)<0$, $f'(x)>0$을 만족시키고 $f(0)=-6$이다. 곡선 $y=f(x)$ 위의 점 $\text{A}(t, f(t))$ (단, $t>0$)에서 $x$축에 내린 수선의 발을 $\text{B}$라 하고, 점 $\text{A}$에서의 접선이 $x$축과 만나는 점을 $\text{C}$라고 하자. 삼각형 $\text{ABC}$의 넓이가 $e^{-3t}$일 때, 곡선 $y=f(x)$와 $x$축, $y$축 및 직선 $x=\ln2$로 둘러싸인 도형의 넓이를 구하여라.} \\

    \exercise{다음 곡선과 $x$축으로 둘러싸인 도형의 넓이를 구하여라.(1)(2)\\
    다음 곡선 또는 직선으로 둘러싸인 도형의 넓이를 구하여라. (3)(4)
     \begin{enumerate}[label=(\arabic*), leftmargin=8mm, nosep]
        \item $y = \sqrt{x^3 - x^2 - 8x + 12}$
        \item $y = e^{-x}\sin x \quad (0 \le x \le 2\pi)$       
        \item $y = e^x\cos x \quad (0 \le x \le \pi), \quad y = e^x\sin x \quad (0 \le x \le \pi), \quad x = 0, \quad x = \pi$

    \end{enumerate}}

    \exercise{오른쪽 그림과 같이 두 곡선

    \exercise{곡선 $y = x\cos x$ (단, $x \ge 0$)가 직선 $y = x$에 접하는 점을 원점에 가까운 순서로 $\text{P}_0$(원점), $\text{P}_1$, $\text{P}_2$, $\cdots$, $\text{P}_{n-1}$, $\text{P}_n$, $\cdots$이라 하고, 자연수 $n$에 대하여 선분 $\text{P}_{n-1}\text{P}_n$과 곡선 $y = x\cos x$로 둘러싸인 도형의 넓이를 $A_n$이라고 하자.
    \begin{enumerate}
        \item[(1)] $A_n$을 $n$으로 나타내어라.
        \item[(2)] $S_n = \sum_{k=1}^{n} A_k$를 $n$으로 나타내어라.
    \end{enumerate}} \\
