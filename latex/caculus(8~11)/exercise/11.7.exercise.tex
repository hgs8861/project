\subsection*{Difficulty: Medium (14 Problems)}
\begin{enumerate}
    \item \textbf{Exercise 9:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{n^2-1}{n^3+1} \]

    \item \textbf{Exercise 10:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{n-1}{n^3+1} \]

    \item \textbf{Exercise 13:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{e^n}{n^2} \]

    \item \textbf{Exercise 16:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{(-1)^n n^4}{4^n} \]

    \item \textbf{Exercise 23:} Determine whether the series is convergent or divergent.
    \[ \sum_{k=1}^{\infty} \frac{2^{k-1} 3^{k+1}}{k^k} \]

    \item \textbf{Exercise 24:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{\sqrt{n^4+1}}{n^3+n} \]

    \item \textbf{Exercise 26:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=2}^{\infty} \frac{(-1)^n}{\sqrt{n}-1} \]

    \item \textbf{Exercise 27:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} (-1)^n \frac{\ln n}{\sqrt{n}} \]

    \item \textbf{Exercise 29:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} (-1)^n \cos(1/n^2) \]

    \item \textbf{Exercise 30:} Determine whether the series is convergent or divergent.
    \[ \sum_{k=1}^{\infty} \frac{1}{2+\sin k} \]

    \item \textbf{Exercise 32:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} n \sin(1/n) \]

    \item \textbf{Exercise 33:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{4-\cos n}{n^{1.5}} \]

    \item \textbf{Exercise 37:} Determine whether the series is convergent or divergent.
    \[ \sum_{k=1}^{\infty} \frac{k \ln k}{(k+1)^3} \]

    \item \textbf{Exercise 38:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{e^{1/n}}{n^2} \]
\end{enumerate}

% --- 난이도 상 (Hard) ---
\subsection*{Difficulty: Hard (8 Problems)}
\begin{enumerate}
    \setcounter{enumi}{14} % 이전 목록 번호 이어서 시작

    \item \textbf{Exercise 5:} Test the series (a) and (b) for convergence or divergence.
    \[ (a)\ \sum_{n=1}^{\infty} \frac{n}{n^2+1} \qquad (b)\ \sum_{n=1}^{\infty} \left( \frac{n}{n^2+1} \right)^n \]

    \item \textbf{Exercise 6:} Test the series (a) and (b) for convergence or divergence.
    \[ (a)\ \sum_{n=1}^{\infty} \frac{\ln n}{n} \qquad (b)\ \sum_{n=10}^{\infty} \frac{1}{n \ln n} \]

    \item \textbf{Exercise 7:} Test the series (a) and (b) for convergence or divergence.
    \[ (a)\ \sum_{n=1}^{\infty} \frac{1}{n+n!} \qquad (b)\ \sum_{n=1}^{\infty} (\frac{1}{n}+\frac{1}{n!}) \]

    \item \textbf{Exercise 21:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{3^n n^2}{n!} \]

    \item \textbf{Exercise 25:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{1 \cdot 3 \cdot 5 \cdot \dots \cdot (2n-1)}{2 \cdot 5 \cdot 8 \cdot \dots \cdot (3n-2)} \]

    \item \textbf{Exercise 28:} Determine whether the series is convergent or divergent.
    \[ \sum_{k=1}^{\infty} \frac{\sqrt[3]{k}-1}{k(\sqrt{k}+1)} \]

    \item \textbf{Exercise 35:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \frac{n!}{e^{n^2}} \]

    \item \textbf{Exercise 43:} Determine whether the series is convergent or divergent.
    \[ \sum_{n=1}^{\infty} \left( \frac{n}{n+1} \right)^{n^2} \]
\end{enumerate}
